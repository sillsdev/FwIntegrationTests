from test_helper import *
from sikuli import *

class ShortCutKeysTest():

    def NavigateDictionaryPane(self):
        find().click()
        assert exists()


    def Findbox1(self):
        wait(1)
        find().below(30).click()
        wait(1)
        #type("f", Key.CTRL)
        keyDown(Key.CTRL)
        type("f")
        keyUp()
        assert exists()
        type(Key.ESC)

    def Delete(self):
        wait(1)
        keyDown(Key.CTRL)
        type(Key.DELETE)
        keyUp(Key.CTRL)
        assert exists()
        type(Key.ESC)

    def UndoRedo(self):
        wait(1)
        keyDown(Key.CTRL)
        type(Key.DELETE)
        keyUp(Key.CTRL)

        find().click()
        wait(, 5)
        type("z", Key.CTRL)
        wait(1)
        assert exists()
        find().highlight(1)
        type("y", Key.CTRL)
        wait(1)
        assert exists()
        find().highlight(1)

    def Findbox2(self):

        keyDown(Key.CTRL)
        type("f", Key.SHIFT)
        keyUp(Key.CTRL)
        assert exists()
        keyDown(Key.ALT)
        keyDown(Key.F4)
        keyUp()
        wait(1)

    def NewEntryBox(self):

        #wait(10)
        type("e", Key.CTRL)
        assert exists()
        type(Key.ESC)

    def LeftandRight(self):
        wait(1)
        find(Pattern().exact()).click()
        find().click()
        find().click()
        find().click()
        find()
        click()
        wait(Pattern().targetOffset(-94,-1))
        find(Pattern().exact()).click()
        find().click()
        wait(2)
        find().click()
        wait(2)

        keyDown(Key.ALT)
        type(Key.LEFT)
        keyUp()

        if exists():
            assert exists(Pattern().similar(0.90))
            find(Pattern().similar(0.90)).highlight(1)
        else:
            print "Right Arrow is not enable it on Texts & Words pane"
            assert exists(Pattern().similar(0.90))

        wait(2)
        keyDown(Key.ALT)
        type(Key.RIGHT)
        keyUp()
        wait(2)
        if exists():
            assert exists(Pattern().similar(0.90))
            find(Pattern().similar(0.90)).highlight(1)
        else:
            assert exists(Pattern().similar(0.90))
            print "Left Arrow is not enable it on Lexicon pane"
